# Scorecard export script (token value stored separately and not printed)
#!/usr/bin/env sh
set -eu
TOKEN=<read from deployment secret file>
OUTDIR=/opt/open-transit-rt/evidence/scorecard
mkdir -p "$OUTDIR"
ts=$(date -u +%Y-%m-%dT%H%M%SZ)
out="$OUTDIR/scorecard-$ts.json"
curl -fsS -X POST http://127.0.0.1:8081/admin/compliance/scorecard \
  -H "Authorization: Bearer $TOKEN" \
  -H 'Content-Type: application/json' \
  --data '{}' > "$out"
sha256sum "$out" > "$out.sha256"
printf '%s exported %s\n' "$(date -u +%Y-%m-%dT%H:%M:%SZ)" "$out" >> "$OUTDIR/scorecard-export-job-history.txt"
